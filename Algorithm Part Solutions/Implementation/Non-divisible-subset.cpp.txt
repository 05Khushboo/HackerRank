    for(ll i=0;i<n;i++) {
    ll b; cin>>b;
        b=b%k;
        a[b]++;
    }
  for(ll i=1;i<=k-1;i++) m+=a[i];
   ll o=1;
    for(ll i=1,j=k-1;i<=k/2;i++,j--)
        { //cout<<i<<" "<<j<<" ";
        if(i==j) {
            if(a[i]!=0) m-=a[i];
            m++;
        }
        else { if(a[i]!=0&&a[j]!=0)
        {c=min(a[i],a[j]);
         m-=c;
        }}
        }

    else {
        if(a[0]!=0)
        o=m+1;
    else o=m;
    }

}
